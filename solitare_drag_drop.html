    <script>

        let draggedCard = null;


        document.addEventListener('dragstart', (e) => {

            draggedCard = e.target;

            e.dataTransfer.setData('text/plain', draggedCard.id);

        });


        document.addEventListener('dragover', (e) => {

            e.preventDefault();

        });


        document.addEventListener('drop', (e) => {

            e.preventDefault();

            const target = e.target;

            if (target.classList.contains('column') || target.classList.contains('foundation') || target.classList.contains('stock')) {

                target.appendChild(draggedCard);

                draggedCard.style.position = 'static';

            }

        });


        function generateCard(cardId, column) {

            const card = document.createElement('div');

            card.className = 'card';

            card.id = cardId;

            card.draggable = true;

            card.innerText = cardId;

            document.getElementById(column).appendChild(card);

        }


        function generateInitialBoard() {

            const board = document.getElementById('board');

            for (let i = 1; i <= 7; i++) {

                const column = document.createElement('div');

                column.className = 'column';

                column.id = `column${i}`;

                column.ondrop = (e) => e.preventDefault();

                board.appendChild(column);

            }

        }


        function generateBoard() {

            const columns = [];

            for (let i = 1; i <= 7; i++) {

                const columnCards = Array.from(document.getElementById(`column${i}`).children)

                    .map(card => card.innerText);

                columns.push(columnCards);

            }


            const foundation = [];

            for (let i = 1; i <= 4; i++) {

                const card = document.getElementById(`foundation${i}`).innerText || '';

                foundation.push(card);

            }


            const stock = Array.from(document.getElementById('stock').children)

                .map(card => card.innerText);


            const boardData = {

                tableau: columns,

                foundation: foundation,

                stock: stock

            };


            document.getElementById('output').innerText = JSON.stringify(boardData, null, 2);

        }


        // Initialize columns

        generateInitialBoard();

    </script>
